# Скрипты на движке 1Script

## Telegram отправка картинки.os

```os
Функция ОбернутьВКавычки(Парам)
	Если Найти(Парам," ")<>0 Тогда
		Возврат """"+Парам+"""";
	Иначе
		Возврат Парам;
	КонецЕсли;
КонецФункции

ПутьККартинке		= "t:\al\";
ИмяФайлаКартинки	= "Alt.jpg";
ПодписьККартинке	= "Тест картинки 2";

ИмяСкрипта	     = "z:\1C\7.7\1Script\bin\Telegram отправка картинки.os";
ИДЧата		     = ЗдесьПодставьтеИДВашегоЧатаКанала;
ИмяСкрипта 	     = ОбернутьВКавычки(ИмяСкрипта);
ИДЧата		     = ОбернутьВКавычки(ИДЧата);
ПутьККартинке 	 = ОбернутьВКавычки(ПутьККартинке);
ИмяФайлаКартинки = ОбернутьВКавычки(ИмяФайлаКартинки);
ПодписьККартинке = ОбернутьВКавычки(ПодписьККартинке);
Команда="z:\1C\7.7\1Script\bin\oscript.exe "+ИмяСкрипта+" "+ИДЧата+" "+ПутьККартинке+" "+ИмяФайлаКартинки+" "+ПодписьККартинке;

//КомандаСистемы(Команда); //ЗапуститьПриложение(Команда);
WshShell=СоздатьОбъект("WScript.Shell");
WshShell.Run(Команда, 0, 0);//3й параметр: -1 - синхронный режим, 0 - асинхронный
```
